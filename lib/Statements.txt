CREATE TABLE Employee
(
   `ID` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
   `FNAME` VARCHAR(40) NOT NULL,
   `LNAME` VARCHAR(40) NOT NULL,
   `SEX` VARCHAR(5) NOT NULL,
   `WITHDRAWAL` INTEGER NOT NULL,
    PRIMARY KEY (ID)
);
